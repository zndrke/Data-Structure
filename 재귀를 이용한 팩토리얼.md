재귀를 이용한 팩토리얼

n!=    

> ​     1         n=0
>
> n*(n-1)!   n>=1



~~~c
#include <stdio.h>

int fact(int n){
	if(n==0)
		return 1;            //n=0일 때
	else 
		return n*fact(n-1);  //n>=1일 때 재귀적으로 fact함수를 다시 호출하여 n-1값 전달
}

int main(void){
	int n;
	scanf("%d",&n);
	printf("%d",fact(n));
	return 0;
}
~~~
